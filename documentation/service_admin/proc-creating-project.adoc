[id='creating-project_{context}']
= Creating the project for {ProductName}

ifeval::["{cmdcli}" == "oc"]
. Create the `enmasse` project:
+
[options="nowrap",subs="attributes"]
----
{cmdcli} new-project enmasse
----
endif::[]
ifeval::["{cmdcli}" == "kubectl"]

. Create the `enmasse` namespace:
+
[options="nowrap",subs="attributes"]
----
{cmdcli} create namespace enmasse
----

. Set the `enmasse` namespace as the default namespace:
+
[options="nowrap",subs="attributes"]
----
kubectl config set-context $(kubectl config current-context) --namespace=enmasse
----
endif::[]
